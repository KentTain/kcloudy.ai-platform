  ALTER TABLE departments
    ADD COLUMN tree_leaf BOOLEAN NOT NULL DEFAULT TRUE,
    ADD COLUMN tree_level INTEGER NOT NULL DEFAULT 0,
    ADD COLUMN tree_sort INTEGER NOT NULL DEFAULT 0,
    ADD COLUMN tree_sorts VARCHAR(128) NOT NULL DEFAULT '',
    ADD COLUMN tree_names VARCHAR(128) NOT NULL DEFAULT '',
    ADD COLUMN parent_ids VARCHAR(1024) NOT NULL DEFAULT '00000000-0000-0000-0000-000000000000,';
