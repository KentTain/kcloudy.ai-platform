# 多租户模块（tenant）设计方案

本文档描述多租户模块的详细设计方案，包括：租户信息的上下文统一注入和获取、不同资源的数据隔离实现。

---

## 1. 基础设施及用途

项目的基础设施：数据库（PostgreSQL）、向量库（pgvector）、Redis（cache、queue、pub/sub）、MinIO，采用 MVC 架构。

---

## 3. 核心协议设计

### 3.1 TenantInfo Protocol

租户信息协议，用于上下文传递和服务间通信。

```python
from dataclasses import dataclass
from typing import Protocol


@dataclass
class TenantDatabaseConfig:
    """租户数据库配置"""
    type: DatabaseType = DatabaseType.POSTGRESQL
    host: str = ""
    port: int = 5432
    database: str = ""
    username: str = ""
    password: str = ""


@dataclass
class TenantStorageConfig:
    """租户存储配置"""
    type: StorageType = StorageType.MINIO
    endpoint: str = ""
    access_key: str = ""
    secret_key: str = ""
    bucket: str = ""


@dataclass
class TenantQueueConfig:
    """租户队列配置"""
    type: QueueType = QueueType.REDIS
    endpoint: str = ""


@dataclass
class TenantPubSubConfig:
    """租户发布订阅配置"""
    type: PubSubType = PubSubType.REDIS
    endpoint: str = ""


class TenantInfo(Protocol):
    """
    租户信息协议

    包含基础信息和可选的资源配置，支持租户级资源隔离。
    """

    # ========== 基础信息 ==========
    @property
    def id(self) -> str: ...

    @property
    def name(self) -> str: ...

    @property
    def code(self) -> str: ...

    @property
    def status(self) -> str: ...

    # ========== 联系人信息（可选）==========
    @property
    def contact_name(self) -> str | None: ...

    @property
    def contact_email(self) -> str | None: ...

    @property
    def contact_phone(self) -> str | None: ...

    # ========== 资源配置（可选）==========
    @property
    def database(self) -> TenantDatabaseConfig | None: ...

    @property
    def storage(self) -> TenantStorageConfig | None: ...

    @property
    def queue(self) -> TenantQueueConfig | None: ...

    @property
    def pubsub(self) -> TenantPubSubConfig | None: ...
```

### 3.2 TenantProvider Protocol

租户提供者协议，抽象租户获取逻辑，支持本地部署和分布式部署。

```python
class TenantProvider(Protocol):

    租户提供者协议

    支持场景：
    - 本地部署：直接数据库访问
    - 分布式部署：通过 RPC/HTTP 调用 IAM 服务

    async def get_tenant(self, tenant_id: str) -> TenantInfo | None:
        """
        获取租户信息

        Args:
            tenant_id: 租户 ID

        Returns:
            TenantInfo | None
        """
        ...

    async def validate_access(self, user_id: str, tenant_id: str) -> bool:
        """
        验证用户是否有权访问租户

        Args:
            user_id: 用户 ID
            tenant_id: 租户 ID

        Returns:
            bool
        """
        ...

    async def get_user_tenants(self, user_id: str) -> list[TenantInfo]:
        """
        获取用户所属的租户列表

        Args:
            user_id: 用户 ID

        Returns:
            list[TenantInfo]
        """
        ...
```

### 3.3 全局注册机制

```python
# framework/tenant/protocols.py

_tenant_provider: TenantProvider | None = None


def register_tenant_provider(provider: TenantProvider) -> None:
    """
    注册租户提供者

    应用启动时调用，注入具体实现。
    """
    global _tenant_provider
    _tenant_provider = provider


def get_tenant_provider() -> TenantProvider:
    """
    获取租户提供者

    Raises:
        RuntimeError: 未注册时抛出
    """
    if _tenant_provider is None:
        raise RuntimeError(
            "TenantProvider not registered. "
            "Call register_tenant_provider() at startup."
        )
    return _tenant_provider
```

### 3.4 SimpleTenant 实现

```python
# framework/tenant/context.py

from dataclasses import dataclass
from typing import Any

from framework.tenant.protocols import (
    TenantInfo,
    TenantDatabaseConfig,
    TenantStorageConfig,
    TenantQueueConfig,
    TenantPubSubConfig,
)


@dataclass
class SimpleTenant:
    """
    简化的租户信息，用于上下文存储

    实现 TenantInfo Protocol。
    """

    # 基础信息
    id: str
    name: str
    code: str
    status: str = "active"

    # 联系人信息（可选）
    contact_name: str | None = None
    contact_email: str | None = None
    contact_phone: str | None = None

    # 资源配置（可选，支持租户级隔离）
    database: TenantDatabaseConfig | None = None
    storage: TenantStorageConfig | None = None
    queue: TenantQueueConfig | None = None
    pubsub: TenantPubSubConfig | None = None

    @classmethod
    def from_model(cls, model: Any) -> "SimpleTenant":
        """从 ORM 模型创建"""
        return cls(
            id=model.id,
            name=model.name,
            code=model.code,
            status=model.status,
            contact_name=getattr(model, "contact_name", None),
            contact_email=getattr(model, "contact_email", None),
            contact_phone=getattr(model, "contact_phone", None),
        )
```

#### IAM 服务实现（本地）

```python
# iam/services/tenant_provider_impl.py

from framework.tenant.protocols import TenantInfo, TenantProvider, SimpleTenant
from iam.models import Tenant, UserTenant
from iam.services.tenant_service import TenantService


class IamTenantProvider(TenantProvider):
    """IAM 模块的 TenantProvider 实现（本地部署）"""

    async def get_tenant(self, tenant_id: str) -> TenantInfo | None:
        tenant = await TenantService.get_by_id(tenant_id)
        if tenant:
            return SimpleTenant.from_model(tenant)
        return None

    async def validate_access(self, user_id: str, tenant_id: str) -> bool:
        async with async_session() as session:
            stmt = select(UserTenant).where(
                UserTenant.user_id == user_id,
                UserTenant.tenant_id == tenant_id,
            )
            result = await session.execute(stmt)
            return result.scalar_one_or_none() is not None

    async def get_user_tenants(self, user_id: str) -> list[TenantInfo]:
        tenants = await TenantService.get_user_tenants(user_id)
        return [SimpleTenant.from_model(t) for t in tenants]
```

## 4. 设计决策

### 4.1 数据库隔离策略

采用字段级隔离（Shared Database, Shared Schema）：

```
┌─────────────────────────────────────────────────────────────┐
│                    数据库隔离策略                            │
├─────────────────────────────────────────────────────────────┤
│  策略：所有租户共享同一数据库，通过 tenant_id 字段隔离数据   │
│                                                             │
│  优点：                                                      │
│  ✅ 实现简单，运维成本低                                     │
│  ✅ 资源共享方便，连接池管理简单                             │
│  ✅ 跨租户查询（管理员场景）容易实现                         │
│                                                             │
│  实现：                                                      │
│  • TenantMixin：模型混入，自动添加 tenant_id 字段            │
│  • TenantEventListener：插入前自动填充 tenant_id             │
│  • TenantQueryInterceptor：查询时自动过滤 tenant_id          │
└─────────────────────────────────────────────────────────────┘
```

### 4.2 Redis 隔离策略

在工具类层面自动注入租户前缀：

```python
# 调用方式（业务代码无感知）
await RedisUtil.set("user:123", data)

# 实际存储（自动添加租户前缀）
# key: "tenant_001:user:123"
```

实现要点：

- `RedisUtil` 内部从 `TenantContext` 获取当前租户 ID
- 所有 key 操作自动拼接前缀 `{tenant_id}:`
- 提供跳过前缀的方法（管理员场景）：`RedisUtil.set(..., skip_tenant=True)`

### 4.3 存储隔离策略

共享 Bucket + 租户目录隔离：

```
┌─────────────────────────────────────────────────────────────┐
│                    MinIO 存储隔离                            │
├─────────────────────────────────────────────────────────────┤
│  bucket: app-files（共享）                                  │
│  ├── tenant_001/                                            │
│  │   ├── avatars/                                          │
│  │   ├── documents/                                        │
│  │   └── temp/                                             │
│  └── tenant_002/                                            │
│      ├── avatars/                                          │
│      ├── documents/                                        │
│      └── temp/                                             │
└─────────────────────────────────────────────────────────────┘
```

实现要点：

- 所有上传操作自动拼接租户目录前缀
- 预签名 URL 生成时包含租户路径
- 提供 API 供管理员跨租户访问

### 4.4 租户缓存策略

采用两级缓存（L1 本地内存 + L2 Redis）：

```
┌─────────────────────────────────────────────────────────────────────────┐
│                         两级缓存架构                                     │
├─────────────────────────────────────────────────────────────────────────┤
│                                                                         │
│   ┌─────────────────┐     Miss     ┌─────────────────┐                 │
│   │   L1 Cache      │ ──────────▶ │   L2 Cache      │                 │
│   │   (本地内存)    │             │   (Redis)       │                 │
│   ├─────────────────┤             ├─────────────────┤                 │
│   │ • 进程级缓存    │             │ • 跨实例共享    │                 │
│   │ • 热点数据      │             │ • TTL: 5 分钟   │                 │
│   │ • 无网络开销    │             │ • 集群同步      │                 │
│   │ • 最大 1000 条  │             │                 │                 │
│   └─────────────────┘             └─────────────────┘                 │
│          │                                │ Miss                      │
│          │                                ▼                           │
│          │                       ┌─────────────────┐                  │
│          │                       │   TenantProvider │                  │
│          │                       │   (IAM 服务)     │                  │
│          │                       └─────────────────┘                  │
│          │                                │                           │
│          └────────────────────────────────┘                           │
│                       写入时双向更新                                   │
│                                                                         │
└─────────────────────────────────────────────────────────────────────────┘
```

#### 缓存流程

```
读取流程:
─────────
1. 检查 L1 本地缓存 → 命中则返回
2. 检查 L2 Redis 缓存 → 命中则回填 L1 并返回
3. 通过 TenantProvider 获取 → 写入 L1、L2 并返回

写入流程:
─────────
1. 更新数据库
2. 更新 L1 本地缓存
3. 更新 L2 Redis 缓存
4. 发布缓存失效消息（其他实例收到后清除 L1）
```

#### Redis Key 设计

```
# 租户信息缓存
tenant:info:{tenant_id}           → Tenant JSON

# 缓存失效通知频道
tenant:cache:invalidate           → {"tenant_id": "xxx", "action": "update|delete"}
```

#### 任务执行器封装

```python
# framework/queue/task_executor.py

from framework.tenant.protocols import get_tenant_provider


class TenantTaskExecutor:
    """租户感知的任务执行器"""

    @staticmethod
    async def execute(message: TaskMessage) -> bool:
        try:
            # 恢复租户上下文
            if message.tenant_id:
                await TenantTaskExecutor._restore_tenant_context(message.tenant_id)

            # 执行任务逻辑
            handler = get_handler(message.task_type)
            await handler(message.payload)
            return True

        finally:
            TenantContext.clear()

    @staticmethod
    async def _restore_tenant_context(tenant_id: str) -> None:
        """通过 TenantProvider 恢复租户上下文"""
        provider = get_tenant_provider()
        tenant = await provider.get_tenant(tenant_id)
        if tenant:
            TenantContext.set_current_tenant(tenant)
```

### 6.2 租户中间件设计

#### 中间件职责

1. **解析租户标识**：从 Header 或 Token 中提取 `tenant_id`
2. **验证租户状态**：通过 TenantProvider 检查租户是否存在、是否激活
3. **验证访问权限**：通过 TenantProvider 检查用户是否有权访问租户
4. **注入租户上下文**：将租户信息存入 `TenantContext`
5. **清理上下文**：请求结束后清理 `TenantContext`

#### 中间件实现

```python
# framework/tenant/middleware.py

from framework.tenant.protocols import get_tenant_provider


class TenantMiddleware(BaseHTTPMiddleware):
    """租户中间件（使用 TenantProvider，无业务模块依赖）"""

    async def dispatch(self, request, call_next):
        try:
            # 1. 解析租户标识
            tenant_id = TenantResolver.resolve(request)
            if not tenant_id:
                raise TenantResolveError("无法解析租户标识")

            # 2. 通过 Provider 获取租户信息
            provider = get_tenant_provider()
            tenant = await provider.get_tenant(tenant_id)

            if not tenant:
                raise TenantNotFoundError(tenant_id)

            # 3. 验证租户状态
            if tenant.status != "active":
                raise TenantInactiveError(tenant_id)

            # 4. 验证访问权限
            user_id = self._get_user_id(request)
            if user_id:
                has_access = await provider.validate_access(user_id, tenant_id)
                if not has_access:
                    raise TenantAccessDeniedError(tenant_id)

            # 5. 注入上下文
            TenantContext.set_current_tenant(tenant)

            # 6. 执行业务逻辑
            return await call_next(request)

        finally:
            TenantContext.clear()
```

### 6.3 租户解析策略

优先级从高到低：

1. **请求头 `X-Tenant-Id`**：用户手动切换租户
2. **Token 中的 `tenant_id`**：用户默认租户
3. **用户主租户**：用户关联的默认租户

```python
def _resolve_tenant_id(request) -> str | None:
    # 优先级1: 请求头
    tenant_id = request.headers.get("X-Tenant-Id")
    if tenant_id:
        return tenant_id

    # 优先级2: Token 中携带
    user = request.state.user
    if user and hasattr(user, "tenant_id"):
        return user.tenant_id

    # 优先级3: 用户主租户（需要查询）
    return None
```
